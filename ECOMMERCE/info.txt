PROGRAMMINO CHE FESTISCE UNA RUBRICA DI CONTATTI E ORDINI DEI CONTATTI

-CREARE IL DB
-CREARE TABELLA CONTATTI
-CREARE TABELLA ORDINI




//CREAZIONE DB SU MYSQL


//CREAZIONE TABELLA contatti

    CREATE TABLE contatti (
        
        id INT AUTO_INCREMENT PRIMARY KEY,
        nome VARCHAR(30) NOT NULL,
        telefono VARCHAR(30) NOT NULL,
        email VARCHAR(30) NOT NULL
        
    );


//CREAZIONE TABELLA ORDINI

    CREATE TABLE ordini (
        id INT AUTO_INCREMENT PRIMARY KEY,
        prodotto VARCHAR(50) NOT NULL,
        quantita INT NOT NULL,
        data_di_ordine DATE NOT NULL,
        
        contatto_id INT NOT NULL,
        FOREIGN KEY (contatto_id) REFERENCES contatti(id) ON DELETE CASCADE -> serve per eliminare tutti gli ordini collegati ad un id specifico nel caso venga cancellato il contatto
    );
